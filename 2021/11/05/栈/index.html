栈

创建最普通的栈

#define MAXSIZE 100
#include<stdio.h>
#include<malloc.h>
#include<stdlib.h>
typedef int Element;//将element定义成所需类型便于修改
struct Stack
{
Element* Data;//存储数据
int Top;//栈顶
int Msz;//最大长度
};
typedef Stack* stack;
stack CreatStack();//创建栈
bool isEmpty(stack S);//判断栈是否为空
bool isFull(stack S);//判断栈是否满
bool Push(stack S, Element x);//将值放入栈
Element Pop(stack S);//将值弹出栈
stack CreatStack()
{
stack S = (stack)malloc(sizeof(Stack));
if (S == NULL)
{
exit(-1);
}
S->Data = (Element*)malloc(sizeof(Element));
S->Msz = MAXSIZE;
S->Top = -1;
return S;
}
bool isEmpty(stack S)
{
return (S->Top == -1);
}
bool isFull(stack S)
{
return (S->Top == S->Msz-1);
}
bool Push(stack S,Element x)
{
if (isFull(S))
{
printf(“栈满”);
return false;
}
else
{
S->Data[++(S->Top)] = x;
return true;
}
}
Element Pop(stack S)
{
if (isEmpty(S))
{
printf(“栈空”);
}
else
{
return S->Data[(S->Top)–];
}
}

用链表创建栈

该栈的特点是可以不局限于长度的创建栈，且能够很好的利用栈的空间

#include<stdio.h>
#include <malloc.h>
#include<stdlib.h>
typedef int Element;//将element定义成所需类型便于修改
struct Stack
{
Element data;//存储数据
Stack* next;//指向下一个内存块
};
typedef Stack* stack;
stack CreatStack();//创建栈
bool isEmpty(stack S);;//判断栈是否为空
void Free(stack S);//释放内存
bool Push(stack S, Element x);//将值放入栈
Element Pop(stack S);//将值弹出栈
stack CreatStack()
{
stack S = (stack)malloc(sizeof(Stack));
if (S == NULL)
{
exit(-1);
}
S->data = 0;
S->next = NULL;
return S;
}
bool isEmpty(stack S)
{
return (S->next==NULL);
}
bool Push(stack S, Element x)
{
stack cell = (stack)malloc(sizeof(Stack));
if (cell == NULL)
{
exit(-1);
}
cell->next = S->next;
cell->data = x;
S->next = cell;
return true;
}
Element Pop(stack S)
{
if (isEmpty(S))
{
printf(“栈空”);
}
else
{
Element a = S->next->data;
stack temp = S->next->next;
free(S->next);
S->next = temp;
return a;
}
}
void Free(stack S)
{
while (S->next != NULL)
{
stack temp = S;
while (temp->next != NULL)
{
temp = temp->next;
}
free(temp);
}
free(S);
}
